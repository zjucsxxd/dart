180 inline bool isSymmetric(const Eigen::MatrixXd& _m, double _tol = 1e-6) {
181  size_t rows = _m.rows();
182  size_t cols = _m.cols();
183 
184  if (rows != cols)
185  return false;
186 
187  for (size_t i = 0; i < rows; ++i) {
188  for (size_t j = i + 1; j < cols; ++j) {
189  if (std::fabs(_m(i, j) - _m(j, i)) > _tol) {
190  std::cout << "A: " << std::endl;
